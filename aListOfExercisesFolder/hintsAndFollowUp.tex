\begin{exercise}
    We can have follow up questions. First compute
    \[
    1+2 = \answer{3}
    \]
    \begin{exercise}
        Knowing that the answer to the previous question is $3$, what is the previous answer plus one?
        \[
        \answer{4}
        \]
    \end{exercise}
\end{exercise}
